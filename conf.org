* Install functions & WSL accommodations
#+begin_src bash
if [ -f "/etc/wsl.conf" ]; then
    IS_WSL=1
else
    IS_WSL=0
fi

install() {
    yay -Sy --needed $*
    echo "Installed $($*)" >> ~/install.log
}

install_native() {
    if [ $IS_WSL -eq 1 ]; then
       echo "ERR: On WSL. Did not install $($*)" >> ~/install.log
       return 0;
    fi

    install $*
    return 0
}

if [ $IS_WSL -eq 1 ]; then
    cat <<EOF >> /etc/wsl.conf

    [interop]
    interop=false
    appendWindowsPath=false
    EOF

    install xclip
fi
#+end_src
